# Use node-gyp for Standard Setup

node-gyp is the standard build tool for Node.js native addons. It handles cross-platform compilation, proper include paths, and linking against the correct Node.js version.

## Why This Matters

- Automatically locates Node.js headers
- Handles platform-specific compilation flags
- Integrates with npm install workflow
- Manages multiple Node.js version targets

## Incorrect

Custom Makefile without node-gyp integration:

```makefile
# BAD: Manual Makefile - breaks on different platforms/Node versions
CXX = g++
CXXFLAGS = -fPIC -shared -I/usr/local/include/node

addon.node: addon.cpp
    $(CXX) $(CXXFLAGS) -o $@ $<

# Problems:
# - Hardcoded include path
# - No Windows support
# - Wrong Node.js version headers
# - Missing N-API flags
```

```cpp
// addon.cpp - will fail to build correctly
#include <node_api.h>  // Where does this come from?

napi_value Init(napi_env env, napi_value exports) {
    return exports;
}

NAPI_MODULE(addon, Init)
```

## Correct

Proper binding.gyp configuration:

```python
# binding.gyp
{
    "targets": [
        {
            "target_name": "addon",
            "sources": [
                "src/addon.cpp",
                "src/worker.cpp"
            ],
            "include_dirs": [
                "<!@(node -p \"require('node-addon-api').include\")"
            ],
            "defines": [
                "NAPI_VERSION=8",
                "NAPI_DISABLE_CPP_EXCEPTIONS"
            ],
            "cflags!": ["-fno-exceptions"],
            "cflags_cc!": ["-fno-exceptions"],
            "conditions": [
                ["OS=='win'", {
                    "defines": [
                        "_HAS_EXCEPTIONS=1"
                    ],
                    "msvs_settings": {
                        "VCCLCompilerTool": {
                            "ExceptionHandling": 1
                        }
                    }
                }],
                ["OS=='mac'", {
                    "xcode_settings": {
                        "GCC_ENABLE_CPP_EXCEPTIONS": "YES",
                        "CLANG_CXX_LIBRARY": "libc++",
                        "MACOSX_DEPLOYMENT_TARGET": "10.15"
                    }
                }]
            ]
        }
    ]
}
```

```json
// package.json
{
    "name": "my-addon",
    "version": "1.0.0",
    "main": "lib/index.js",
    "scripts": {
        "install": "node-gyp rebuild",
        "build": "node-gyp build",
        "rebuild": "node-gyp rebuild",
        "clean": "node-gyp clean"
    },
    "dependencies": {
        "node-addon-api": "^7.0.0"
    },
    "devDependencies": {
        "node-gyp": "^10.0.0"
    },
    "gypfile": true
}
```

```cpp
// src/addon.cpp
#include <napi.h>

Napi::Value HelloWorld(const Napi::CallbackInfo& info) {
    Napi::Env env = info.Env();
    return Napi::String::New(env, "Hello from native addon!");
}

Napi::Object Init(Napi::Env env, Napi::Object exports) {
    exports.Set(
        Napi::String::New(env, "hello"),
        Napi::Function::New(env, HelloWorld)
    );
    return exports;
}

NODE_API_MODULE(addon, Init)
```

## Common binding.gyp Patterns

```python
# Multiple source files
{
    "targets": [{
        "target_name": "addon",
        "sources": [
            "src/addon.cpp",
            "src/utils.cpp",
            "src/worker.cpp"
        ]
    }]
}

# External library linking
{
    "targets": [{
        "target_name": "addon",
        "sources": ["src/addon.cpp"],
        "libraries": [
            "-lssl",
            "-lcrypto"
        ],
        "conditions": [
            ["OS=='linux'", {
                "libraries": ["-L/usr/local/lib"]
            }],
            ["OS=='mac'", {
                "libraries": [
                    "-L/opt/homebrew/lib",
                    "-L/usr/local/opt/openssl/lib"
                ],
                "include_dirs": [
                    "/opt/homebrew/include",
                    "/usr/local/opt/openssl/include"
                ]
            }]
        ]
    }]
}

# Debug build configuration
{
    "targets": [{
        "target_name": "addon",
        "sources": ["src/addon.cpp"],
        "configurations": {
            "Debug": {
                "defines": ["DEBUG", "_DEBUG"],
                "cflags": ["-g", "-O0"],
                "xcode_settings": {
                    "GCC_OPTIMIZATION_LEVEL": "0"
                }
            },
            "Release": {
                "defines": ["NDEBUG"],
                "cflags": ["-O3"],
                "xcode_settings": {
                    "GCC_OPTIMIZATION_LEVEL": "3"
                }
            }
        }
    }]
}
```

## Build Commands

```bash
# Standard build
npm install

# Rebuild from scratch
npm run rebuild

# Build with debug symbols
node-gyp rebuild --debug

# Build for specific Node version
node-gyp rebuild --target=18.0.0

# Verbose output for debugging
node-gyp rebuild --verbose

# Clean build artifacts
node-gyp clean
```

## Verifying the Build

```javascript
// lib/index.js
const path = require('path');

let addon;
try {
    // Try release build first
    addon = require('../build/Release/addon.node');
} catch (e) {
    // Fall back to debug build
    addon = require('../build/Debug/addon.node');
}

module.exports = addon;

// test/test.js
const addon = require('../lib');
console.log(addon.hello()); // "Hello from native addon!"
```
